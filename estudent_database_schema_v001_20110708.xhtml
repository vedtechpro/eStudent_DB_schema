Table address
id bigserial NOT NULL
city varchar( 50 )
kifle_ketema varchar( 50 )
kebele varchar( 50 )
woreda varchar( 50 )
house_numebr varchar( 50 )
telephone varchar( 50 )
person_id bigserial foreign key for person
Indexes
pk_address primary key ( id)
pk_address_0 unique ( person_id)


Table countries
id bigserial NOT NULL
name varchar( 255 ) NOT NULL
code varchar( 10 ) NOT NULL
continent varchar( 255 )
Indexes
pk_countries primary key ( id)


Table courses
id bigserial NOT NULL
code varchar( 50 ) NOT NULL Course number (Course code) is a unique-human-readable identifier that is used to uniquely represent a course. Format : ABC-XXXX => Comp234
title varchar( 255 ) Course title is a descriptive text that is used to give additional information about the course
credit int2 NOT NULL Credit given to this course interms of contact hours
description text A textual description about the course
Indexes
pk_courses primary key ( id)
Constraints
ck_unique_course_number unique(course_number)


Table people
id bigserial NOT NULL
first_name varchar( 255 ) NOT NULL
father_name varchar( 255 ) NOT NULL
grandfather_name varchar( 255 )
sex varchar( 1 ) NOT NULL DEFO 'F'
birth_date timestamp NOT NULL
nationality int8 NOT NULL DEFO 1
birth_place varchar( 60 ) Birth Place of a person
Indexes
pk_people primary key ( id)
idx_people ( nationality)
Foreign Keys
fk_people_countries ( nationality ) ref countries ( id )
fk_people ( id ) ref address ( person_id )
Constraints
ck_unique_person unique(first_name,father_name,grandfather_name) Check uniqueness of a person by checking fullname


Table prerequisites
Course prerequisites
course_id int8 NOT NULL
prerequisite_id int8 NOT NULL
Indexes
idx_prerequisites ( course_id)
idx_prerequisites ( prerequisite_id)
idx_prerequisites_0 primary key ( course_id, prerequisite_id)
Foreign Keys
fk_prerequisites_courses ( course_id ) ref courses ( id )
fk_prerequisites_courses_prerequisite ( prerequisite_id ) ref courses ( id )
Constraints
ck_unique_course_set unique(course_id,prerequisite_id)


Table students
A table for students.
id bigserial NOT NULL Primary key and internal identifier
person_id bigserial NOT NULL
id_number varchar( 255 ) Id Number of a student
department_id bigserial foreign key for department
year int8 current year
dorm_id bigserial foreign key for Dorm
campus_id bigserial foreign key for campus
joined_date int4 The date that the student joined the university
Indexes
pk_students primary key ( id) Campus Name, where the student is enrolled at
idx_students ( person_id)
Foreign Keys
fk_students_people ( person_id ) ref people ( id )
Constraints
ck_unque_person_in_students unique(person_id)
